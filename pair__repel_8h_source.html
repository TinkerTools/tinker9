69 // dmpik(9) == dmpik[4]
70 // dmpik(11) == dmpik[5]

76 real dkr = dot3(dk, dr);
77 real3 qk_dr = matvec(qkxx, qkxy, qkxz, qkyy, qkyz, qkzz, dr);
78 real qkr = dot3(dr, qk_dr);
79

81 real qik = dot3(qi_dr, qk_dr);
82 real diqk = dot3(di, qk_dr);
83 real dkqi = dot3(dk, qi_dr);

88 real term2 = valk * dir - vali * dkr + dik;
89 real term3 = vali * qkr + valk * qir - dir * dkr + 2 * (dkqi - diqk + qiqk);
90 real term4 = dir * qkr - dkr * qir - 4 * qik;
91 real term5 = qir * qkr;

101 term5 * dmpik[5];
102 term1 = -valk * dmpik[1] + dkr * dmpik[2] - qkr * dmpik[3];
103 term2 = vali * dmpik[1] + dir * dmpik[2] + qir * dmpik[3];
104 term3 = 2 * dmpik[2];
105 term4 = 2 * (-valk * dmpik[2] + dkr * dmpik[3] - qkr * dmpik[4]);
106 term5 = 2 * (-vali * dmpik[2] - dir * dmpik[3] - qir * dmpik[4]);

114 real qky = qk_dr.y;
115 real qkz = qk_dr.z;

118 real qiyk = qixy * qkx + qiyy * qky + qiyz * qkz;
119 real qizk = qixz * qkx + qiyz * qky + qizz * qkz;
120 real qkxi = qkxx * qix + qkxy * qiy + qkxz * qiz; // |Qk Qi r>

128 real dkqiy = dk.x * qixy + dk.y * qiyy + dk.z * qiyz;

134 frc0.y = de * dr.y + term1 * di.y + term2 * dk.y + term3 * (diqky - dkqiy) + term4 * qiy +
135 term5 * qky + term6 * (qiyk + qkyi);
136 frc0.z = de * dr.z + term1 * di.z + term2 * dk.z + term3 * (diqkz - dkqiz) + term4 * qiz +
137 term5 * qkz + term6 * (qizk + qkzi);

144 pgrad.frcy *= sizik;
